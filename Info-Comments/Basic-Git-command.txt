1) git status   // Shows the current state of the working directory and staging area (modified, staged, untracked files)

2) git pull   // Fetches changes from the remote repository and merges them into your current branch

3.1) git add <file-name>    // Adds a specific file to the staging area  
3.2) git add .              // Adds all changed and untracked files to the staging area

4) git commit -m "Your commit message"   // Saves the staged changes to the local repository with a descriptive message

5) git push   // Sends committed changes from the local repository to the remote repository
